List of examples that demostrates:
- How to customize the overall theme of Push UI Kit.
- How to customize the theme for light and dark modes.

```jsx live
// customPropMinimized='true'
// Import necessary components from @pushchain/ui-kit
import {
  PushUniversalWalletProvider,
  PushUniversalAccountButton,
  usePushWalletContext,
  PushUI,
} from '@pushchain/ui-kit';

function App() {
  const walletConfig = {
    network: PushUI.CONSTANTS.PUSH_NETWORK.TESTNET,
    login: {
      email: true,
      google: true,
      wallet: {
        enabled: true,
      },
      appPreview: true,
    },
    modal: {
      loginLayout: PushUI.CONSTANTS.LOGIN.SPLIT,
      connectedLayout: PushUI.CONSTANTS.CONNECTED.HOVER,
      appPreview: true,
    },
  };

  const appMetadata = {
    logoUrl:
      'https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D',
    title: 'Test App Title',
    description: 'Test App Description',
  };

  function WalletUI() {
    const { universalAccount } = usePushWalletContext();

    return (
      <div>
        <PushUniversalAccountButton />
      </div>
    );
  }

  return (
    <PushUniversalWalletProvider
      config={walletConfig}
      app={appMetadata}
      themeOverrides={{
        '--pw-core-bg-primary-color': '#FF474C',
        '--pw-core-font-family': 'Inter',
        '--pw-core-text-primary-color': '#fff',
        '--pw-core-text-secondary-color': '#F5F6F8',
        '--pw-core-btn-border-radius': '15px',
      }}
    >
      <WalletUI />
    </PushUniversalWalletProvider>
  );
}
```

```jsx live
// customPropMinimized='true'
import {
  PushUniversalWalletProvider,
  PushUniversalAccountButton,
  usePushWalletContext,
  PushUI,
} from '@pushchain/ui-kit';

function App() {
  const walletConfig = {
    network: PushUI.CONSTANTS.PUSH_NETWORK.TESTNET,
    login: {
      email: true,
      google: true,
      wallet: {
        enabled: true,
      },
      appPreview: true,
    },
    modal: {
      loginLayout: PushUI.CONSTANTS.LOGIN.SPLIT,
      connectedLayout: PushUI.CONSTANTS.CONNECTED.HOVER,
      appPreview: true,
    },
  };

  const appMetadata = {
    logoUrl:
      'https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D',
    title: 'Test App Title',
    description: 'Test App Description',
  };

  function WalletUI() {
    const { universalAccount } = usePushWalletContext();

    return (
      <div>
        <PushUniversalAccountButton />
      </div>
    );
  }

  return (
    <PushUniversalWalletProvider
      config={walletConfig}
      app={appMetadata}
      themeMode={PushUI.CONSTANTS.THEME.LIGHT}
      themeOverrides={{
        '--pw-core-bg-primary-color': 'red',
        light: {
          '--pw-core-bg-primary-color': 'yellow',
        },
        dark: {
          '--pw-core-bg-primary-color': 'green',
        },
      }}
    >
      <WalletUI />
    </PushUniversalWalletProvider>
  );
}
```
