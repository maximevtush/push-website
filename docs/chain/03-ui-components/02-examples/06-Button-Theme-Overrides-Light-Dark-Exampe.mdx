# Example

This example demonstrates a basic wallet integration with the button theme overrides for both light and dark modes.

## Live playground

```jsx live
// customPropMinimized='true'
import {
  PushUniversalWalletProvider,
  PushUniversalAccountButton,
  usePushWalletContext,
  CONSTANTS,
} from '@pushchain/ui-kit';

function App() {
  const walletConfig = {
    env: CONSTANTS.ENV.DEVNET,
    login: {
      email: true,
      google: true,
      wallet: {
        enabled: true,
      },
      appPreview: true,
    },
    modal: {
      loginLayout: CONSTANTS.LOGIN.SPLIT,
      connectedLayout: CONSTANTS.CONNECTED.HOVER,
      appPreview: true,
    },
  };

  const appMetadata = {
    logoUrl:
      'https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D',
    title: 'Test App Title',
    description: 'Test App Description',
  };

  function WalletUI() {
    const { universalAddress } = usePushWalletContext();

    return (
      <div>
        <PushUniversalAccountButton
          themeOverrides={{
            light: {
              '--pwauth-btn-connect-text-color': 'red',
              '--pwauth-btn-connect-bg-color	': '#fff',
              '--pwauth-btn-connected-text-color': '#000',
              '--pwauth-btn-connected-bg-color': '#000000',
            },
            dark: {
              '--pwauth-btn-connect-text-color': 'blue',
              '--pwauth-btn-connect-bg-color': '#000',
              '--pwauth-btn-connected-text-color': '#fff',
              '--pwauth-btn-connected-bg-color': '#ffffff',
            },
          }}
        />
      </div>
    );
  }

  return (
    <PushUniversalWalletProvider
      config={walletConfig}
      app={appMetadata}
      themeMode={CONSTANTS.THEME.DARK}
    >
      <WalletUI />
    </PushUniversalWalletProvider>
  );
}
```
