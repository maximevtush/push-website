## Example

This example demonstrates a multiple wallet integration with the Push Universal Wallet.

## Live playground

```jsx live
// customPropMinimized = 'true'
import {
  PushUniversalWalletProvider,
  PushUniversalAccountButton,
  usePushWalletContext,
  CONSTANTS,
} from '@pushchain/ui-kit';

function App() {
  const walletConfig = {
    uid: 'wallet1',
    env: CONSTANTS.ENV.DEVNET,
    login: {
      email: true,
      google: true,
      wallet: {
        enabled: true,
      },
      appPreview: false,
    },
    modal: {
      loginLayout: CONSTANTS.LOGIN.SPLIT,
      connectedLayout: CONSTANTS.CONNECTED.HOVER,
      appPreview: true,
    },
  };

  const secondWalletConfig = {
    uid: 'wallet2',
    env: CONSTANTS.ENV.DEVNET,
    login: {
      email: false,
      google: false,
      wallet: {
        enabled: true,
      },
      appPreview: true,
    },
  };

  const appMetadata = {
    logoUrl:
      'https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D',
    title: 'Test App Title',
    description: 'Test App Description',
  };

  const secondAppMetadata = {
    logoUrl:
      'https://plus.unsplash.com/premium_photo-1746731481770-08b2f71661d0?q=80&w=2671&auto=format&fit=crop&ixlib=rb-4.1.0&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D',
    title: 'Test App Title',
    description: 'Test App Description',
  };

  function WalletUI() {
    const { universalAddress } = usePushWalletContext('wallet1');
    const { universalAddress: secondAddress } = usePushWalletContext('wallet2');

    return (
      <div
        style={{
          display: 'flex',
          flexDirection: 'row',
          alignItems: 'center',
          justifyContent: 'center',
          gap: '2rem',
        }}
      >
        <PushUniversalAccountButton uid='wallet1' />
        <PushUniversalAccountButton uid='wallet2' />
      </div>
    );
  }

  return (
    <PushUniversalWalletProvider config={walletConfig} app={appMetadata}>
      <PushUniversalWalletProvider
        config={secondWalletConfig}
        app={secondAppMetadata}
      >
        <WalletUI />
      </PushUniversalWalletProvider>
    </PushUniversalWalletProvider>
  );
}
```
